
So what I did eventually was:
1) Download officedeploymenttool_6831-5775.exe
2) Run -- Setup.exe /download "configuration.xml"  (this will download the full Office to your install folder)
3) Run -- Setup.exe /configure "configuration.xml"  (this will install all components listed in your config.xml file)
 
Here are three config files that I made:
 
Full office install including full repair. (this will fix Skype for Business and keep the Office License, all components, including Outlook OSTs).
 
<Configuration>
   <Add SourcePath="c:\odt" OfficeClientEdition="32" >
   <Product ID="O365ProPlusRetail">
      <Language ID="en-us" />
    </Product>
  </Add> 
</Configuration>
 
Lync install only, with no Office components already installed.
 
<Configuration>
    <Add SourcePath="c:\odt" OfficeClientEdition="32" Channel="Current">
        <Product ID="O365ProPlusRetail">
            <Language ID="en-us"/>
            <ExcludeApp ID="Excel"/>
            <ExcludeApp ID="Groove"/>
            <ExcludeApp ID="OneNote"/>
            <ExcludeApp ID="OneDrive"/>
            <ExcludeApp ID="Outlook"/>
            <ExcludeApp ID="PowerPoint"/>
            <ExcludeApp ID="Project"/>
            <ExcludeApp ID="Publisher"/>
            <ExcludeApp ID="Visio"/>
            <ExcludeApp ID="Word"/>
            <ExcludeApp ID="Access"/>
        </Product>
    </Add>
</Configuration>
 
 
Remove all Office components.
 
<Configuration>
    <Remove All="TRUE"/>
</Configuration>
